Script inside Angular Template

January 1, 2014 by Reza

By default AngularJs doesn't support script inside template. To achieve this feature we can write a simple directive

Let's say your template contain following script


    < script template-with-script>
        console.log('from tpl');
        alert('from template')
    < /script>

Your directive looks like

        
    app.directive('templateWithScript', function(){
        return{
            link: function (scope, element) {
                scope.$on('$includeContentLoaded', function(){
                    var script = document.createElement('script');
                    //have to be careful regarding innerHtml
                    script.innerHTML = element[0].innerHTML;
                    document.body.appendChild(script);
                })

            }
        }
    })
